// Bubble Sort
#include <array>
#include <print>

void sort(auto& input) {
    for (unsigned it = input.size(); it != 0; --it) {
        unsigned i1 = 0;
        unsigned i2 = 1;
        for (; i2 != it; ++i1, ++i2) {
            auto& in1 = input[i1];
            auto& in2 = input[i2];
            if (in1 > in2) {
                std::swap(in1, in2);
            }
        }
    }
}

int main() {
    std::array input{1, 8, 4, 6, 3, 7, 0, 9, 6, 8, 3, 7, 5, 6};
    sort(input);

    std::print("{}", input);
}
